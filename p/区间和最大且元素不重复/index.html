class Solution:
    def minDeletions(self, s: str) -> int:
        # ‰ªéÂ§ßÂà∞Â∞èÊéíÂ∫è
        # arr[i-1] = min(arr[i-1], arr[i]-1)
        temp = [0] * 26
        for c in s:
            temp[ord(c)-ord('a')] += 1
        
        # ÊéíÂ∫è
        temp.sort(reverse=True)
        ans = 0
        for i in range(1, len(temp)):
            a = temp[i]

            temp[i] = min(temp[i-1]-1, temp[i])
            if temp[i] <= 0:
                temp[i] = 0
            ans += a - temp[i]

        return ans
