private ListNode mergeTwoLists(ListNode a, ListNode b) {
    ListNode curA = a;
    ListNode curB = b;
    ListNode head = new ListNode(0);
    ListNode tail = head;
    while (curA != null && curB != null) {
        if (curA.val < curB.val) {
            tail.next = curA;
            curA = curA.next;
        } else {
            tail.next = curB;
            curB = curB.next;
        }
        tail = tail.next;
    }
    tail.next = curA == null ? curB : curA;
    return head.next;
}
